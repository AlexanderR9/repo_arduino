
//подключение модуля к ардуино
Подключается к шине i2c (пины A4 и A5)
    DS3231     Arduino
     GND   ->   GND
     VCC   ->   VCC
     SDA   ->   A4
     SCL   ->   A5


// Подключаем библиотеку для работы с модулем
#include <iarduino_RTC.h>  

//создание объекта, type елемент множества типов RTC модулей, может принимать следующие значения:
//RTC_DS1302, RTC_DS1307, RTC_DS3231, необходимо выбрать соответствующий своей модели
iarduino_RTC my_obj(type); 

//в функции setup необходимо выполнить инициализацию объекта 
Serial.begin(9600); //открытие COM порта для обмена
my_obj.begin(); // Инициализация RTC модуля

//Установка времени модуля:
my_obj.settime(0, 51, 21, 27, 10, 15, 2); // 0 сек, 51 мин, 21 час, 27, октября, 2015 года, вторник

//Получение времени и вывод на монитор:
uint8_t D, M, Y, h, m, s, W;  //предварительно объявляем переменные, в которые предстоит записывать дату и время
my_obj.gettime();  // Считываем текущее время из модуля.
    D = my_obj.day;      // записать в переменную день месяца 1-31.
    M = my_obj.month;    // записать в переменную номер месяца 1-12.
    Y = my_obj.year;     // записать в переменную значение года 0-99.
    h = my_obj.Hours;    // записать в переменную текущий час 0-23.
    m = my_obj.minutes;  // записать в переменную текущую минуту 0-59.
    s = my_obj.seconds;  // записать в переменную текущую секунду 0-59.
    W = my_obj.weekday;  // записать в переменную день недели 0-6.
Serial.println((String) D+"-"+M+"-"+Y+", "+h+":"+m+":"+s+", "+W); // Выводим время на монитор, одной строкой.
Serial.println(my_obj.gettime("d-m-Y, H:i:s, D")); // выводим время в монитор, одной строкой (2-й способ)




//////////////////необязательные функции////////////////////////////
Функция period(): - установка минимального периода обращения к модулю в минутах (от 0 до 255).
Функция может быть полезна, если шина передачи данных сильно нагружена другими устройствами.
Пример: my_obj.period(1); // Указываем обращаться к модулю RTC не чаще чем один раз в минуту (обычно устанавлевается в setup)
